## 画布状态管理与组件交互实现

### 1 Redux 状态设计

#### 1.1 状态结构设计

**设计原则**：

| 原则           | 说明                                      |
| -------------- | ----------------------------------------- |
| **单一数据源** | 所有组件状态存储在唯一的 store 中         |
| **不可变性**   | 状态只能通过 reducer 更新，保证可追踪性   |
| **范式化**     | 使用 `byId` + `allIds` 结构，避免嵌套数据 |
| **按领域分割** | 状态按业务领域划分，减少耦合              |

**完整状态结构**：

```typescript
// 根状态类型

  canvas: CanvasState;
  editor: EditorState;

  clipboard: ClipboardState;
  ui: UIState;

  // 选中状态

    selectedIds: string[];

    isDragging: boolean;

  snapEnabled: boolean;
  panelWidth: {
    left: number;
    right: number;
  };
}
```

#### 1.2 数据范式化

**为什么需要范式化？**

```plain
┌──────────────────────────────────────────────────────────────────┐
│                    范式化 vs 嵌套数据对比                          │
└──────────────────────────────────────────────────────────────────┘

  非范式化（嵌套）：
  ┌─────────────────────────────────────────────────────────────┐
  │  component: {                                               │
  │    id: '1',                                                │
  │    children: [                                             │
  │      { id: '2', children: [...] },                        │
  │      { id: '3', children: [...] }                         │
  │    ]                                                        │
  │  }                                                          │
  └─────────────────────────────────────────────────────────────┘
      问题：
      • 更新深层子组件需要复制整个树
      • 引用不稳定，组件重新渲染
      • 状态管理复杂

  范式化（扁平）：
  ┌─────────────────────────────────────────────────────────────┐
  │  byId: {                                                    │
  │    '1': { id: '1', childIds: ['2', '3'] },                 │
  │    '2': { id: '2', parentId: '1' },                        │
  │    '3': { id: '3', parentId: '1' }                        │
  │  },                                                         │
  │  allIds: ['1', '2', '3']                                   │
  └─────────────────────────────────────────────────────────────┘
      优势：
      • 更新任意组件只需更新一条记录
      • 引用稳定，减少不必要渲染
      • 便于缓存和计算
```

#### 1.4 Reducer 实现

```typescript
// canvasReducer.ts
const initialState: CanvasState = {
  components: {
    byId: {},
    allIds: [],
    rootId: '',
  },
  selection: {
    selectedIds: [],
    hoveredId: null,
    editingId: null,
  },
  drag: {
    isDragging: false,
    sourceId: null,
    targetId: null,
    position: { x: 0, y: 0 },
  },
  clipboard: {
    type: null,
    componentIds: [],
  },
};

function canvasReducer(state = initialState, action: CanvasAction): CanvasState {
  switch (action.type) {
    // 选择组件
    case 'SELECT_COMPONENT':
      return {
        ...state,
        selection: {
          ...state.selection,
          selectedIds: action.payload.additive
            ? [...state.selection.selectedIds, action.payload.id]
            : [action.payload.id],
        },
      };

    // 添加组件
    case 'ADD_COMPONENT': {
      const { component, parentId, index } = action.payload;
      const parent = state.components.byId[parentId];

      // 更新 byId
      const newById = {
        ...state.components.byId,
        [component.id]: {
          ...component,
          parentId,
        },
      };

      // 更新父组件的 childIds
      const childIds = parent?.childIds || [];
      childIds.splice(index, 0, component.id);
      newById[parentId] = {
        ...parent,
        childIds: [...childIds],
      };

      return {
        ...state,
        components: {
          ...state.components,
          byId: newById,
          allIds: [...state.components.allIds, component.id],
        },
      };
    }

    // 更新组件属性
    case 'UPDATE_COMPONENT': {
      const { id, changes } = action.payload;
      const component = state.components.byId[id];

      if (!component) return state;

      return {
        ...state,
        components: {
          ...state.components,
          byId: {
            ...state.components.byId,
            [id]: {
              ...component,
              ...changes,
            },
          },
        },
      };
    }

    // 删除组件
    case 'DELETE_COMPONENT': {
      const { id } = action.payload;
      const component = state.components.byId[id];

      if (!component) return state;

      // 递归收集所有待删除的 ID
      const idsToDelete = collectDescendantIds(id, state.components.byId);

      // 从 allIds 中移除
      const newAllIds = state.components.allIds.filter(
        (cid) => !idsToDelete.includes(cid)
      );

      // 从父组件的 childIds 中移除
      const parent = state.components.byId[component.parentId];
      const newById = {
        ...state.components.byId,
        [component.parentId]: {
          ...parent,
          childIds: parent.childIds.filter((cid) => cid !== id),
        },
      };

      // 删除所有后代组件
      idsToDelete.forEach((cid) => {
        delete newById[cid];
      });

      return {
        ...state,
        components: {
          ...state.components,
          byId: newById,
          allIds: newAllIds,
        },
        selection: {
          ...state.selection,
          selectedIds: state.selection.selectedIds.filter(
            (sid) => !idsToDelete.includes(sid)
          ),
        },
      };
    }

    // 移动组件（排序）
    case 'MOVE_COMPONENT': {
      const { sourceId, targetId, position } = action.payload;
      const source = state.components.byId[sourceId];
      const target = state.components.byId[targetId];

      if (!source || !target) return state;

      // 从原位置移除
      const sourceParent = state.components.byId[source.parentId];
      const newSourceChildIds = sourceParent.childIds.filter(
        (cid) => cid !== sourceId
      );

      // 插入新位置
      const targetIndex = position === 'before'
        ? target.parentId === source.parentId
          ? newSourceChildIds.indexOf(targetId)
          : newSourceChildIds.length
        : target.parentId === source.parentId
          ? newSourceChildIds.indexOf(targetId) + 1
          : newSourceChildIds.length + 1;

      newSourceChildIds.splice(targetIndex, 0, sourceId);

      return {
        ...state,
        components: {
          ...state.components,
          byId: {
            ...state.components.byId,
            [source.parentId]: {
              ...sourceParent,
              childIds: newSourceChildIds,
            },
            [sourceId]: {
              ...source,
              parentId: source.parentId,
            },
          },
        },
      };
    }

    default:
      return state;
  }
}
```

```typescript
const handleDragStart = (event: DragStartEvent) => {
  const { active } = event;

  dispatch({
    type: 'SET_DRAG_SOURCE',
    payload: { sourceId: active.id as string },
  });

  dispatch({
    type: 'SELECT_COMPONENT',
    payload: { id: active.id, additive: false },
  });

// 拖拽过程中
const handleDragOver = (event: DragOverEvent) => {
  const { active, over } = event;

  if (!over) return;

  // 计算放置位置
  const isBefore = calculateIsBefore(active, over);

  dispatch({
    type: 'SET_DROP_TARGET',
    payload: {
      targetId: over.id as string,
      position: isBefore ? 'before' : 'after',
    },
  });
};

// 拖拽结束
const handleDragEnd = (event: DragEndEvent) => {
  const { active, over } = event;

  dispatch({
    type: 'CLEAR_DRAG',
  });

  if (!over) return;

  // 执行组件移动
  if (active.id !== over.id) {
    const isBefore = calculateIsBefore(active, over);

    dispatch({
      type: 'MOVE_COMPONENT',
      payload: {
        sourceId: active.id as string,
        targetId: over.id as string,
        position: isBefore ? 'before' : 'after',
      },
    });
  }
};
```

#### 3 核心算法

```typescript
// 计算放置位置（在目标之前还是之后）
function calculateIsBefore(
  active: DraggableSyntheticPointer,
  over: DraggableSyntheticPointer
): boolean {
  const activeRect = active.rect.current.translated;
  const overRect = over.rect.current.translated;

  // 比较中心点 Y 坐标
  const activeCenterY = activeRect.top + activeRect.height / 2;
  const overCenterY = overRect.top + overRect.height / 2;

  if (activeCenterY < overCenterY) {
    return true;  // 放在目标之前
  } else if (activeCenterY > overCenterY) {
    return false;  // 放在目标之后
  } else {
    // Y 相同，比较 X 坐标
    const activeCenterX = activeRect.left + activeRect.width / 2;
    const overCenterX = overRect.left + overRect.width / 2;
    return activeCenterX < overCenterX;
  }
}

// 计算组件在列表中的排序
function computeSortedIds(
  allIds: string[],
  byId: Record<string, Component>
): string[] {
  // 使用拓扑排序确保父子顺序
  const sorted: string[] = [];

  function visit(id: string) {
    const component = byId[id];
    if (!component) return;

    sorted.push(id);

    component.childIds?.forEach(childId => {
      visit(childId);
    });
  }

  // 从根节点开始
  allIds.forEach(id => {
    if (!byId[id].parentId) {
      visit(id);
    }
  });

  return sorted;
}
```

#### 3.2 层级操作实现

```typescript
// 层级 reducer
function layerReducer(state: CanvasState, action: LayerAction): CanvasState {
  switch (action.type) {
    case 'BRING_TO_FRONT': {
      const { id } = action;
      const allIds = [...state.components.allIds];
      const index = allIds.indexOf(id);

      if (index === -1) return state;

      // 移除并添加到末尾
      allIds.splice(index, 1);
      allIds.push(id);

      return updateZIndexes(state, allIds);
    }

    case 'SEND_TO_BACK': {
      const { id } = action;
      const allIds = [...state.components.allIds];
      const index = allIds.indexOf(id);

      if (index === -1) return state;

      // 移除并添加到开头
      allIds.splice(index, 1);
      allIds.unshift(id);

      return updateZIndexes(state, allIds);
    }

    case 'BRING_FORWARD': {
      const { id } = action;
      const allIds = [...state.components.allIds];
      const index = allIds.indexOf(id);

      if (index === -1 || index === allIds.length - 1) return state;

      // 与后一个交换
      [allIds[index], allIds[index + 1]] = [allIds[index + 1], allIds[index]];

      return updateZIndexes(state, allIds);
    }

    case 'SEND_BACKWARD': {
      const { id } = action;
      const allIds = [...state.components.allIds];
      const index = allIds.indexOf(id);

      if (index <= 0) return state;

      // 与前一个交换
      [allIds[index], allIds[index - 1]] = [allIds[index - 1], allIds[index]];

      return updateZIndexes(state, allIds);
    }

    default:
      return state;
  }
}

// 更新所有组件的 z-index
function updateZIndexes(state: CanvasState, orderedIds: string[]): CanvasState {
  const newById = { ...state.components.byId };

  orderedIds.forEach((id, index) => {
    newById[id] = {
      ...newById[id],
      zIndex: index + 1,  // z-index 从 1 开始
    };
  });

  return {
    ...state,
    components: {
      ...state.components,
      byId: newById,
    },
  };
}
```

```typescript
function useSyncProperties() {
  const selectedIds = useSelector(
    (state: RootState) => state.canvas.selection.selectedIds
  );

  const [properties, setProperties] = useState<Record<string, any>>({});

  useEffect(() => {
    if (selectedIds.length === 1) {
      // 单选时获取选中组件的属性
      const component = getComponentById(selectedIds[0]);
      setProperties(component?.props || {});
    } else if (selectedIds.length > 1) {
      // 多选时计算共同属性
      const commonProps = computeCommonProps(selectedIds);
      setProperties(commonProps);
    } else {
      setProperties({});
    }
  }, [selectedIds]);

  return properties;
}
```

### 5 表单联动方案

#### 5.1 表单联动场景

| 联动类型     | 场景描述                     | 示例                   |
| ------------ | ---------------------------- | ---------------------- |
| **显示联动** | 根据条件显示/隐藏字段        | 商品有折扣时显示原价   |
| **值联动**   | 修改某字段自动更新其他字段   | 修改数量自动计算总价   |
| **选项联动** | 选择某选项后更新其他选项列表 | 选择省份后更新城市列表 |
| **验证联动** | 根据条件改变验证规则         | 必填项根据类型变化     |

#### 5.2 实现方案

```typescript
// 表单联动配置
interface FormCondition {
  field: string;
  operator: 'eq' | 'neq' | 'gt' | 'lt' | 'in' | 'exists';
  value: any;
}

interface FormLinkage {
  type: 'show' | 'hide' | 'require' | 'value';
  condition: FormCondition;
  target: string;
  action: any;
}

// 联动引擎
class FormLinkageEngine {
  private linkages: FormLinkage[] = [];
  private formValues: Record<string, any>;

  register(linkages: FormLinkage[]) {
    this.linkages = linkages;
  }

  evaluate(formValues: Record<string, any>): FormLinkageResult {
    this.formValues = formValues;

    const result: FormLinkageResult = {
      visibleFields: new Set(Object.keys(formValues)),
      requiredFields: new Set<string>(),
      fieldValues: { ...formValues },
    };

    for (const linkage of this.linkages) {
      const isConditionMet = this.evaluateCondition(linkage.condition);

      if (isConditionMet) {
        this.applyLinkage(linkage, result);
      }
    }

    return result;
  }

  private evaluateCondition(condition: FormCondition): boolean {
    const fieldValue = this.formValues[condition.field];

    switch (condition.operator) {
      case 'eq':
        return fieldValue === condition.value;
      case 'neq':
        return fieldValue !== condition.value;
      case 'gt':
        return Number(fieldValue) > Number(condition.value);
      case 'lt':
        return Number(fieldValue) < Number(condition.value);
      case 'in':
        return condition.value.includes(fieldValue);
      case 'exists':
        return fieldValue !== undefined && fieldValue !== null;
      default:
        return false;
    }
  }

  private applyLinkage(linkage: FormLinkage, result: FormLinkageResult) {
    switch (linkage.type) {
      case 'show':
        result.visibleFields.add(linkage.target);
        break;
      case 'hide':
        result.visibleFields.delete(linkage.target);
        break;
      case 'require':
        result.requiredFields.add(linkage.target);
        break;
      case 'value':
        result.fieldValues[linkage.target] = linkage.action;
        break;
    }
  }
}

// 使用示例
const linkages: FormLinkage[] = [
  {
    type: 'show',
    condition: { field: 'hasDiscount', operator: 'eq', value: true },
    target: 'originPrice',
    action: null,
  },
  {
    type: 'require',
    condition: { field: 'type', operator: 'eq', value: 'product' },
    target: 'sku',
    action: null,
  },
];
```

```typescript
// 表单联动 Hook
function useFormLinkage(schema: FormSchema, initialValues: Record<string, any>) {
  const [values, setValues] = useState(initialValues);
  const [touched, setTouched] = useState<Record<string, boolean>>({});

  const engine = useMemo(() => {
    const linkageEngine = new FormLinkageEngine();
    linkageEngine.register(schema.linkages || []);
    return linkageEngine;
  }, [schema]);

  const result = useMemo(() => {
    return engine.evaluate(values);
  }, [engine, values]);

  const handleChange = useCallback((field: string, value: any) => {
    setValues((prev) => ({ ...prev, [field]: value }));
    setTouched((prev) => ({ ...prev, [field]: true }));
  }, []);

  return {
    values: result.fieldValues,
    visibleFields: result.visibleFields,
    requiredFields: result.requiredFields,
    handleChange,
    errors: validate(values, schema, result.requiredFields),
  };
}
```
