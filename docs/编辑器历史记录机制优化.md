##  编辑器历史记录机制优化

### 1 问题分析

#### 1.1 原历史记录机制功能缺陷

**原始快照模式的问题**：

| 问题类型       | 具体表现                 | 影响程度 |
| -------------- | ------------------------ | -------- |
| **内存占用高** | 每次操作保存完整状态快照 | 严重     |
| **响应延迟**   | 大文档保存/恢复耗时长    | 严重     |
| **功能单一**   | 只支持整体 Undo/Redo     | 中等     |
| **无法合并**   | 连续相同操作无法合并     | 中等     |
| **不稳定**     | 复杂嵌套场景回退异常     | 严重     |

**原始实现分析**：

### 2 设计模式应用

#### 2.1 Command Pattern 核心概念

**为什么选择 Command Pattern？**

```plain
┌──────────────────────────────────────────────────────────────────┐
│                    设计模式对比分析                                │
└──────────────────────────────────────────────────────────────────┘

  ┌─────────────────────────────────────────────────────────────┐
  │  Memento Pattern (快照模式)                                  │
  │  ├─ 优点：实现简单，恢复完整                                 │
  │  ├─ 缺点：内存占用大，无法部分恢复                          │
  │  └─ 适用：状态简单、频率低的场景                            │
  └─────────────────────────────────────────────────────────────┘

  ┌─────────────────────────────────────────────────────────────┐
  │  Command Pattern (命令模式) - 选中                          │
  │  ├─ 优点：内存占用小，支持合并，可部分撤销                   │
  │  ├─ 缺点：实现复杂，命令类膨胀                              │
  │  └─ 适用：复杂编辑器、频繁操作场景                          │
  └─────────────────────────────────────────────────────────────┘

  ┌─────────────────────────────────────────────────────────────┐
  │  Transaction Pattern (事务模式)                             │
  │  ├─ 优点：支持批量操作，原子性                             │
  │  ├─ 缺点：开销大，不适合细粒度操作                          │
  │  └─ 适用：需要事务保证的场景                                │
```

#### 2.2 命令接口设计

```typescript
// 命令接口定义
interface Command {
  // 执行命令
  execute(): void;

  // 撤销命令
  undo(): void;

  // 重做命令
  redo(): void;

  // 获取命令描述（用于调试/展示）
  getDescription(): string;

  // 命令合并（可选）
  merge?(other: Command): Command | null;
}

// 命令基类
abstract class BaseCommand implements Command {
  protected dispatch: AppDispatch;
  protected timestamp: number;

  constructor(dispatch: AppDispatch) {
    this.dispatch = dispatch;
    this.timestamp = Date.now();
  }

  abstract execute(): void;
  abstract undo(): void;
  abstract redo(): void;
  abstract getDescription(): string;

  // 默认实现：不可合并
  merge?(other: Command): Command | null {
    return null;
  }
}
```

**添加组件命令**：

```typescript
// AddComponentCommand.ts
class AddComponentCommand extends BaseCommand {
  private component: Component;
  private parentId: string;
  private index: number;
  private executedComponentId: string | null = null;

  constructor(
    dispatch: AppDispatch,
    component: Component,
    parentId: string,
    index: number
  ) {
    super(dispatch);
    this.component = cloneDeep(component);
    this.parentId = parentId;
    this.index = index;
  }

  execute(): void {
    // 如果是 redo，确保生成相同的 component id
    if (this.executedComponentId) {
      this.dispatch(
        addComponentWithId({
          component: {
            ...this.component,
            id: this.executedComponentId,
          },
          parentId: this.parentId,
          index: this.index,
        })
      );
    } else {
      // 首次执行，生成新 ID
      this.executedComponentId = this.component.id;
      this.dispatch(
        addComponent({
          component: this.component,
          parentId: this.parentId,
          index: this.index,
        })
      );
    }
  }

  undo(): void {
    if (this.executedComponentId) {
      this.dispatch(
        removeComponent({
          id: this.executedComponentId,
        })
      );
    }
  }

  redo(): void {
    this.execute();
  }

  getDescription(): string {
    return `添加组件: ${this.component.type}`;
  }

  merge(other: Command): AddComponentCommand | null {
    // 连续的添加组件操作不合并（保持各自独立）
    return null;
  }
}
```

```typescript
// UpdatePropsCommand.ts
class UpdatePropsCommand extends BaseCommand {
  private componentId: string;
  private oldProps: Record<string, any>;
  private newProps: Record<string, any>;
  private tempProps: Record<string, any> | null = null;

  constructor(
    dispatch: AppDispatch,
    componentId: string,
    oldProps: Record<string, any>,
    newProps: Record<string, any>
  ) {
    super(dispatch);
    this.componentId = componentId;
    this.oldProps = cloneDeep(oldProps);
    this.newProps = cloneDeep(newProps);
  }

  execute(): void {
    this.tempProps = cloneDeep(this.newProps);
    this.dispatch(
      updateComponentProps({
        id: this.componentId,
        props: this.newProps,
      })
    );
  }

  undo(): void {
    this.dispatch(
      updateComponentProps({
        id: this.componentId,
        props: this.oldProps,
      })
    );
  }

  redo(): void {
    if (this.tempProps) {
      this.dispatch(
        updateComponentProps({
          id: this.componentId,
          props: this.tempProps,
        })
      );
    }
  }

  getDescription(): string {
    return `修改属性`;
  }

  // 核心：可合并连续的属性修改
  merge(other: Command): UpdatePropsCommand | null {
    // 必须是同类命令
    if (!(other instanceof UpdatePropsCommand)) {
      return null;
    }

    // 必须是同一组件
    if (other.componentId !== this.componentId) {
      return null;
    }

    // 合并为最新的 newProps
    return new UpdatePropsCommand(
      this.dispatch,
      this.componentId,
      this.oldProps,  // 保留原始 oldProps，用于完整回退
      other.newProps  // 使用最新的 newProps
    );
  }
}
```

```typescript
// DeleteComponentCommand.ts
class DeleteComponentCommand extends BaseCommand {
  private component: Component;
  private parentId: string;
  private index: number;
  private childComponents: Component[] = [];

  constructor(
    dispatch: AppDispatch,
    componentId: string,
    componentsById: Record<string, Component>
  ) {
    super(dispatch);

    // 保存被删除组件的完整信息
    const component = componentsById[componentId];
    this.component = cloneDeep(component);
    this.parentId = component.parentId;

    // 收集所有子组件
    this.collectChildren(componentId, componentsById);
  }

  private collectChildren(id: string, byId: Record<string, Component>) {
    const component = byId[id];
    if (!component) return;

    this.childComponents.push(cloneDeep(component));

    component.childIds?.forEach((childId) => {
      this.collectChildren(childId, byId);
    });
  }

  execute(): void {
    this.dispatch(
      removeComponent({
        id: this.component.id,
        includeChildren: true,
      })
    );
  }

  undo(): void {
    // 恢复组件及其子组件
    this.dispatch(
      restoreComponents({
        components: [this.component, ...this.childComponents],
        parentId: this.parentId,
      })
    );
  }

  redo(): void {
    this.execute();
  }

  getDescription(): string {
    return `删除组件: ${this.component.type}`;
  }
}
```

#### 2 合并规则定义

```typescript
// 合并策略接口
interface MergeStrategy {
  // 判断两个命令是否可合并
  canMerge(prev: Command, next: Command): boolean;

  // 执行合并
  merge(prev: Command, next: Command): Command;
}

// 更新属性合并策略
const UpdatePropsMergeStrategy: MergeStrategy = {
  canMerge(prev: Command, next: Command): boolean {
    if (!(prev instanceof UpdatePropsCommand && next instanceof UpdatePropsCommand)) {
      return false;
    }
    // 同一组件的属性修改可合并
    return prev.componentId === next.componentId;
  },

  merge(prev: UpdatePropsCommand, next: UpdatePropsCommand): UpdatePropsCommand {
    // 复用 UpdatePropsCommand 的 merge 方法
    return prev.merge(next) as UpdatePropsCommand;
  }
};

// 拖拽移动合并策略
const DragMoveMergeStrategy: MergeStrategy = {
  private lastMergeTime: number = 0;
  private mergeWindow: number = 300;  // 300ms 内可合并

  canMerge(prev: Command, next: Command): boolean {
    if (!(prev instanceof MoveComponentCommand && next instanceof MoveComponentCommand)) {
      return false;
    }

    // 同一组件的移动可合并
    if (prev.componentId !== next.componentId) {
      return false;
    }

    // 在时间窗口内
    const now = Date.now();
    return now - this.lastMergeTime < this.mergeWindow;
  },

  merge(prev: MoveComponentCommand, next: MoveComponentCommand): MoveComponentCommand {
    this.lastMergeTime = Date.now();
    // 返回新的移动命令，起点为 prev 的终点
    return new MoveComponentCommand(
      next.dispatch,
      next.componentId,
      prev.toParentId,      // 起点为 prev 的终点
      prev.toIndex,
      next.toParentId,
      next.toIndex
    );
  }
};
```

#### 3 合并执行流程

```typescript
// 合并执行流程
function tryMerge(manager: HistoryManager, newCommand: Command): Command {
  const lastCommand = manager.getLastCommand();

  if (!lastCommand) {
    return newCommand;
  }

  // 遍历所有合并策略
  for (const strategy of mergeStrategies) {
    if (strategy.canMerge(lastCommand, newCommand)) {
      // 执行合并
      return strategy.merge(lastCommand, newCommand);
    }
  }

  // 不可合并，返回新命令
  return newCommand;
}
```

### 4 回溯稳定性保障

#### 4.1 异常处理机制

```typescript
// 稳定性保障设计
class StableHistoryManager {
  private past: Command[] = [];
  private future: Command[] = [];
  private maxSize: number = 50;

  // 异常捕获包装
  execute(command: Command): boolean {
    try {
      // 记录当前状态用于回滚
      const currentState = store.getState();

      // 执行命令
      command.execute();

      // 验证状态有效性
      if (!this.validateState()) {
        // 状态无效，回滚
        this.rollback(currentState);
        return false;
      }

      // 合并命令
      const mergedCommand = this.tryMerge(command);

      if (mergedCommand !== command) {
        // 替换最后一个命令
        this.past[this.past.length - 1] = mergedCommand;
      } else {
        // 添加新命令
        this.past.push(command);
      }

      // 清空未来栈
      this.future = [];

      // 限制历史记录数量
      this.trim();

      return true;
    } catch (error) {
      console.error('Command execution failed:', error);
      return false;
    }
  }

  // 撤销操作
  undo(): boolean {
    try {
      const command = this.past.pop();
      if (!command) return false;

      command.undo();

      if (!this.validateState()) {
        // 回滚失败，尝试重做
        command.redo();
        this.past.push(command);
        return false;
      }

      this.future.push(command);
      return true;
    } catch (error) {
      console.error('Undo failed:', error);
      return false;
    }
  }

  // 状态验证
  private validateState(): boolean {
    const state = store.getState();

    // 验证组件 ID 唯一性
    const ids = new Set();
    for (const id of state.canvas.components.allIds) {
      if (ids.has(id)) {
        console.error('Duplicate component ID:', id);
        return false;
      }
      ids.add(id);
    }

    // 验证父子关系一致性
    for (const [id, component] of Object.entries(state.canvas.components.byId)) {
      if (component.parentId) {
        const parent = state.canvas.components.byId[component.parentId];
        if (!parent) {
          console.error('Parent not found for component:', id);
          return false;
        }
        if (!parent.childIds.includes(id)) {
          console.error('Parent-child relationship broken:', id);
          return false;
        }
      }
    }

    return true;
  }
}
```

#### 4.2 边界情况处理

| 边界情况         | 问题描述                | 解决方案                     |
| ---------------- | ----------------------- | ---------------------------- |
| **空撤销栈**     | 无可撤销操作时调用 undo | 返回 false，不报错           |
| **连续撤销**     | 快速多次调用 undo       | 加锁防止并发                 |
| **命令执行失败** | 命令执行抛出异常        | 捕获异常，状态回滚           |
| **状态校验失败** | 命令执行后状态异常      | 拒绝添加，保留原状态         |
| **内存溢出**     | 历史记录过多            | 限制 maxSize，超出时移除最旧 |
| **组件被删除**   | 撤销时组件已被删除      | 先恢复组件，再执行撤销       |

```typescript
// 边界情况处理示例
class HistoryManager {
  private isUndoing: boolean = false;
  private lock: boolean = false;

  async undo(): Promise<boolean> {
    // 防止并发
    if (this.lock || this.isUndoing) {
      return false;
    }

    this.lock = true;
    this.isUndoing = true;

    try {
      // 检查撤销栈
      if (this.past.length === 0) {
        return false;
      }

      // 执行撤销
      return this.executeUndo();
    } finally {
      this.lock = false;
      this.isUndoing = false;
    }
  }

  private executeUndo(): boolean {
    try {
      const command = this.past.pop();
      if (!command) return false;

      // 如果是被删除组件的撤销命令，先恢复组件
      if (command instanceof DeleteComponentCommand) {
        this.ensureComponentExists(command);
      }

      command.undo();
      this.future.push(command);

      return true;
    } catch (error) {
      console.error('Undo execution error:', error);
      return false;
    }
  }

  // 确保组件存在（用于撤销删除操作）
  private ensureComponentExists(command: DeleteComponentCommand): void {
    const state = store.getState();
    const componentId = command.getComponentId();

    if (!state.canvas.components.byId[componentId]) {
      // 组件已被其他操作删除，撤销失败
      throw new Error('Component no longer exists');
    }
  }
}
```
